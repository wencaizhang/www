2 POST 请求

以 form-data 形式提交数据

1
2
3
4

9

postData (url, payload) {
  return fetch(url, {
    method: 'POST',
    body: JSON.stringify(payload),
    headers: new Headers({
      'Content-Type': 'application/x-www-form-urlencoded'
    })
  })
}

    headers: new Headers({
      'Content-Type': 'application/json'
    })
  })
}
